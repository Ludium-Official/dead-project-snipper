// Use DBML to define your database structure

Table Projects {
  project_id UUID PK
  project_name Varchar
  created_date varchar
  recent_action_date datetime
}

Table XAccounts {
  project_id UUID
  user_id INT pk
  user_name VARCHAR
}

Table XFollowers {
  followers_id VARCHAR pk
  user_id INT
}

Table XActivityLog {
  tweet_id INT pk
  account_id INT
  like_count INT
  reply_count INT
  retweet_count INT
  views INT
  tweet VARCHAR
  timeparsed VARCHAR
}
  
Table GithubRepos {
  project_id INT
  user_id INT
  repo_id INT pk
}

Table GithubCommitLog {
  github_repo_id FK
  commit_count int
  issue_count INT
  PR_count INT
  watcher_count INT
  datetime TIMESTAMP
}

Table FundedMembers {
  project_id FK
  name Varchar
}

Table NearAddress {
  project_id FK
  near_address Varchar Unique
}

Table OnChainActivityLog {
  near_address FK
  Activity_log varchar
}


Ref: XAccounts.project_id > Projects.project_id

Ref: XFollowers.user_id > XAccounts.user_id

Ref: XActivityLog.account_id > XAccounts.user_id

Ref: GithubRepos.project_id > Projects.project_id

Ref: GithubCommitLog.github_repo_id > GithubRepos.repo_id

Ref: NearAddress.project_id > Projects.project_id

Ref: FundedMembers.project_id > Projects.project_id

Ref: OnChainActivityLog.near_address > NearAddress.near_address
