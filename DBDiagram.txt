Table Projects {
  project_id UUID PK
  project_name Varchar
  created_date varchar
  recent_action_date datetime
}

Table XAccounts (
  project_id UUID FK,
  user_id INT pk,
  user_name VARCHAR
);

Table XFollowers (
  followers_id VARCHAR pk,
  user_id INT FK
)

Table XActivityLog (
  tweet_id INT pk,
  account_id INT FK,
  like_count INT,
  reply_count INT,
  retweet_count INT,
  views INT,
  tweet VARCHAR,
  timeparsed VARCHAR
);

Table GithubRepos (
  project_id INT FK,
  project_id INT,
  user_id INT,
  repo_id INT pk
);

Table GithuCommitLog (
  github_repo_id FK,
  commit_count INT,
  issue_count INT,
  PR_count INT,
  watcher_count INT,
  datetime TIMESTAMP
);

Table FundedMembers {
  project_id FK
  name Varchar
}

Table NearAddress {
  project_id FK
  near_address Varchar Unique
}

Table OnChainActivityLog {
  near_address FK
  Activity_log varchar
}


Ref: NearAddress.project_id > Projects.project_id

Ref: FundedMembers.project_id > Projects.project_id

Ref: XAccounts.project_id > Projects.project_id

Ref: GithubRepos.project_id > Projects.project_id

Ref: GithubRepoActivityLog.github_repo > GithubRepos.github_repo

Ref: XActivityLog.account_id > XAccounts.account_id

Ref: OnChainActivityLog.near_address > NearAddress.near_address
